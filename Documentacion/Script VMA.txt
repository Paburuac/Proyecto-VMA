-- Verificación: cantidad de productos por categoria.
SELECT c.nombre_categoria, COUNT(p.id_producto) AS cantidad
FROM categoria c
LEFT JOIN producto p ON p.id_categoria = c.id_categoria
GROUP BY c.nombre_categoria
ORDER BY cantidad DESC, c.nombre_categoria;


-- Muestra de cada categoria y los productos dentro de ellas

SELECT 
    c.nombre_categoria,
    p.descripcion
FROM producto p
JOIN categoria c 
    ON p.id_categoria = c.id_categoria
ORDER BY c.nombre_categoria, p.descripcion;
